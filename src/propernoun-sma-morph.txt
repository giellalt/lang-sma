! Continuation lexica for personal names
! --------------------------------------


! At the moment, these continuation lexica are identical. But since I
! suspect that linguistic input will prove that the words behave
! differently, they are kept distinct.

! Names from Sweden

LEXICON NAME ! these are the non-vow-final names from Sweden
CNAME_EVEN ;

! Vowel final names
! - - - - - - - - -

LEXICON VNAME_ODD ! Names ending in stressless vowel
+N+Prop+Sg: N_ODD_SG ;
+N+Prop+Pl: N_ODD_PL ;
+N+Prop:    N_ODD_ESS ;
+N+Prop+Attr: FINAL1 ;

LEXICON VNAME_EVEN ! Names ending in stressless vowel
+N+Prop+Sg: NIEJTE_SG ;
+N+Prop+Pl: NIEJTE_PL ;
+N+Prop:    EVEN_ESS ;
+N+Prop+Attr: FINAL1 ;
