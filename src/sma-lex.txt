LEXICON R
 #+Use/Circ:# Noun ;                                    ! This is the normal case.		         
 #@P.NeedNoun.ON@+Use/Circ:#@P.NeedNoun.ON@ Verb ;      ! Flag diacr declares that the compound   
 #@P.NeedNoun.ON@+Use/Circ:#@P.NeedNoun.ON@ Adjective ; ! needs a N tag later in the derivation  
 +Hyph#+Use/Circ:%- Noun ;                        ! These 3 cases add a hyphen, so     
 +Hyph#+Use/Circ:%-@P.NeedNoun.ON@  Verb ;        ! Sem-julggaštus and maana-gåetie	      
 +Hyph#+Use/Circ:%-@P.NeedNoun.ON@  Adjective ;   ! are allowed.			              

 %-#+Use/Circ:%-# Acronym ;    ! maana-tv, "lomme-cd-spelar"
 %-#+Use/Circ:%-# ProperNoun ; ! Proper nouns as the 2nd part of compounds for non-hyph.
                               ! words. viessu-London goes through here.

 +Cmpnd:%- ENDLEX ;     ! This to take care of Oahpo- ja dutkandept - want this in speller


LEXICON RHyph ! Recursive lexicon from all classes REQUIRING a hyphen to follow.
              ! NBNBNB!!! The hyphen should be added here! NOT in the other lexicons!

 %-#+Use/Circ:%-# Noun;                                     ! This is the normal case.
 %-#@P.NeedNoun.ON@+Use/Circ:%-#@P.NeedNoun.ON@ Verb;       ! Flag diacr declares that the compound
 %-#@P.NeedNoun.ON@+Use/Circ:%-#@P.NeedNoun.ON@ Adjective ; ! needs a N tag later in the derivation

 %-#+Use/Circ:%-# Acronym ;    ! maana-tv, "lomme-cd-spelar"
 %-#+Use/Circ:%-# ProperNoun ; ! Proper nouns as the 2nd part of compounds for hyph-words.
                               ! London-Hull is covered here, whereas Hull-viessu
                               ! is covered by RHyph + Noun.

 +Cmpnd:%- ENDLEX ;      ! This to take care of Oahpo- ja dutkandept - want this in speller

! Proper noun compounds without hyphens are NOT presently covered - if they were
! to be added, they need to be marked +Use/Sub. Examples: LondonHull & viessuLondon.
