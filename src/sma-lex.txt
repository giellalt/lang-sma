 # Noun ;                           ! This is the normal case.		         ^C^
 #@P.NeedNoun.ON@  Verb ;      ! Flag diacr declares that the compound   ^C^
 #@P.NeedNoun.ON@  Adjective ; ! needs a +N tag later in the derivation  ^C^
 +Hyph#:%- Noun ;                        ! These 3 cases add a hyphen, so     ^C^
 +Hyph#:%-@P.NeedNoun.ON@  Verb ;        ! Sem-julggaštus and the like	      ^C^
 +Hyph#:%-@P.NeedNoun.ON@  Adjective ;   ! is allowed.			              ^C^
! # ProperNoun ;     ! Now, let us allow proper nouns as the 2nd part of com- ^C^
 #:%- ProperNoun ;  ! pounds. Thus: London-Hull, LondonHull, Hull-viessu      ^C^
                    ! and viessuLondon are now all ok. Some refinement could  ^C^
                    ! perhaps be done, but corpus research is needed.	      ^C^
 +Cmpnd:%- # ;     ! This to take care of Oahpo- ja dutkandept - want this in speller
